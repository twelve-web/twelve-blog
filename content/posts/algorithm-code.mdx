## 二分查找
```js
function dichotomySearch(arr,target){
    let low =0
    let high =arr.length-1
    while(low<=high){
    let mid = Math.floor((low + high) /2) 
        if(target==arr[mid]){
            return mid
        }else if(target>arr[mid]) {
            low = mid +1
        }else{
            high =mid -1
        }
    }
    return -1
}
let arr =[1,3,5,6,7,8,9,9]
console.log(dichotomySearch(arr,5))
```
